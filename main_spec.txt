# -*- mode: python ; coding: utf-8 -*-

# Import hàm tiện ích từ PyInstaller
from PyInstaller.utils.hooks import collect_data_files

# --- BƯỚC 1: Thu thập dữ liệu từ các thư viện ---
# Thu thập tất cả các file dữ liệu (non-code) từ thư viện ultralytics
# và gán vào biến ultralytics_data.
ultralytics_data = collect_data_files('ultralytics')

# Thêm các tệp dữ liệu tùy chỉnh của bạn (ví dụ: model .pt, file config, hình ảnh...)
# Cú pháp: ('đường/dẫn/đến/file/của/bạn.pt', 'thư/mục/đích/trong/app')
# Dấu '.' có nghĩa là thư mục gốc của ứng dụng sau khi đóng gói.
custom_data = [
    ('best.pt', '.'),
    ('config/my_config.yaml', 'config')
]


# --- BƯỚE 2: Cấu hình PyInstaller Analysis ---
block_cipher = None

a = Analysis(
    ['main.py'],  # Tên file script chính của bạn
    pathex=[],
    binaries=[],
    # Khởi tạo `datas` với các tệp tùy chỉnh của bạn
    datas=custom_data,
    hiddenimports=[
        'ultralytics',
        'torch',
        'torchvision',
        'cv2',
        'numpy',
        'PIL',
        'yaml',
        'pandas',
        'tqdm'
    ],
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[],
    win_no_prefer_redirects=False,
    win_private_assemblies=False,
    cipher=block_cipher,
    noarchive=False,
)


a.datas += ultralytics_data


# --- BƯỚC 4: Cấu hình build ---
pyz = PYZ(a.pure, a.zipped_data, cipher=block_cipher)

exe = EXE(
    pyz,
    a.scripts,
    a.binaries,
    a.zipfiles,
    a.datas, # a.datas giờ đã bao gồm cả custom_data và ultralytics_data
    [],
    name='ten_ung_dung_cua_ban',
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=True,
    upx_exclude=[],
    runtime_tmpdir=None,
    console=True,  # Đặt là False nếu là ứng dụng có giao diện đồ họa (GUI)
    disable_windowed_traceback=False,
    argv_emulation=False,
    target_arch=None,
    codesign_identity=None,
    entitlements_file=None,
)
